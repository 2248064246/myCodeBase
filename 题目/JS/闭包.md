

# 闭包题目

```javaScript
function fn1(n, o) {
  console.log(o)
  return {
    fn1: function (m) {
      return fn1(m, n)
    }
  }
}
var a = fn1(0) //? => {fn1: function}  undefined
a.fn1(1) //? =>   0 {fn1: function}
a.fn1(2) //? =>   0
a.fn1(3) //?  =>   0
var b = fn1(0).fn1(1).fn1(2).fn1(3) // ? undefined 0 1 2
var c = fn1(0).fn1(1) //? undefined 0
c.fn1(2) //? 1
c.fn1(3) //? 1
fn1(1, 2) // => 2
```
